import React from "react";
import { useState } from "react";
import { Typography, Box, CssBaseline, Button, Card, CardActions, CardContent, CardMedia, FormControl, FilledInput, TextField, Input } from "@mui/material";
import axios from 'axios';

const Contact = () => {
    const [name, setName] = useState("");
    const [email, setEmail] = useState("");
    const [message, setMessage] = useState("");

    const handleSubmit = (event) => {
        event.preventDefault();
        axios.post('/submit', { name, email, message })
            .then(response => console.log(response))
            .catch(error => console.log(error));
    }

    return (
        <Box variant="div" sx={{ display: "flex", justifyContent: "space-evenly", height: "auto" , backgroundColor: "#050505" , padding: "70px" , marginBottom: "2px"}}>
            <Box sx={{ marginTop: "20px" }}>
                <Typography variant="h2" sx={{}}>Contact Me</Typography>
                <form onSubmit={handleSubmit}>
                    <TextField sx={{ margin: "10px", width: "400px" , borderRadius: "10px"  }}
                      id="name" 
                      label="Name" 
                      variant="outlined"
                      type="text"
                      value={name}
                      onChange={(e) => setName(e.target.value)}
                    />
                    <TextField sx={{ margin: "10px", width: "400px" , borderRadius: "10px" }} 
                      id="email" 
                      label="Email" 
                      variant="outlined" 
                      type="email"
                      value={email}
                      onChange={(e) => setEmail(e.target.value)}
                    />
                    <TextField sx={{ margin: "10px", width: "400px" , borderRadius: "10px" }} 
                      id="message" 
                      label="Message" 
                      variant="outlined" 
                      type="text"
                      multiline
                      value={message}
                      onChange={(e) => setMessage(e.target.value)}
                    />
                    <Button type="submit" variant="contained" sx={{ width: "400px" , margin: "10px",  backgroundColor: "#01537e" , borderRadius: "10px" }}>
                      <Typography variant="h5" sx={{ color: "whitesmoke"}}>Send</Typography>
                    </Button> 
                </form>
            </Box>
            <img
              src="https://res.cloudinary.com/dkrinlbmx/image/upload/v1682617535/Portfolio/vecteezy_3d-business-mail-send-icon-or-3d-business-important-mail_10915785_381_avecnd.png"
              alt="message illustration"
              width="400px"
            />
        </Box>
    )
}

export default Contact;
